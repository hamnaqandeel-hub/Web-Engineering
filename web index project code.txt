        function handleRoleClick(el) {
            if (!el) return;
            
            document.querySelectorAll('.profile-card').forEach(c => c.classList.remove('clicked'));
            el.classList.add('clicked');
           
            currentRole = el.id.includes('teacher') ? 'teacher' : 'student';
           
            setTimeout(() => {
                const roleSelection = document.getElementById('roleSelection');
                if (roleSelection) {
                    roleSelection.style.display = 'none';
                }
                
                if(currentRole === 'teacher') {
                    const teacherLoginBox = document.getElementById('teacherLoginBox');
                    if (teacherLoginBox) {
                        teacherLoginBox.style.display = 'block';
                    }
                } else {
                    const studentLoginBox = document.getElementById('studentLoginBox');
                    if (studentLoginBox) {
                        studentLoginBox.style.display = 'block';
                    }
                }
            }, 400);
        }

        window.showTeacherSection = function(sectionId) {
            if (!sectionId) return;
            
            // 1. Saare sections se 'active' class hatayein
            const sections = document.querySelectorAll('#teacherPortal .section');
            sections.forEach(section => section.classList.remove('active'));
            
            // 2. Saare nav-links se 'active' class hatayein
            const links = document.querySelectorAll('#teacherPortal .nav-link');
            links.forEach(link => link.classList.remove('active'));
            
            // 3. Target section ko active karein
            const section = document.getElementById(sectionId);
            if (section) {
                section.classList.add('active');
            }
            
            // 4. Event check ko safe banayein
            if (typeof event !== 'undefined' && event !== null && event.target) {
                if (event.target.classList) {
                    event.target.classList.add('active');
                }
            }

            // 5. Different sections ke liye specific actions
            if(sectionId === 'coursesView') {
                if (typeof window.loadCourses === 'function') {
                    window.loadCourses();
                }
            } else if(sectionId === 'attendanceView') {
                // Date set karein
                const today = new Date().toISOString().split('T')[0];
                const dateInput = document.getElementById('attnDate');
                if (dateInput) {
                    dateInput.value = today;
                }
                
                // Dropdown load karein
                if (typeof window.loadAttendanceCourses === 'function') {

                }
            } else if(sectionId === 'quizView') {
                if (typeof window.loadCoursesToDropdown === 'function') {

                // Course dropdown load karein

                                }
                            }
                        });
                    }
                }
            }

        window.loadStudentsForAttendance = async function(courseId) {
            if (!courseId) return;
            
            const tbody = document.getElementById('attendanceTableBody');
            const date = document.getElementById('attnDate')?.value;
            
            if (!tbody) return;
            
            tbody.innerHTML = '<tr><td colspan="4">Loading attendance data...</td></tr>';
            
            try {
                const courseDoc = await getDoc(doc(db, "courses", courseId));
                if (!courseDoc.exists()) {
                    tbody.innerHTML = '<tr><td colspan="4">Course not found</td></tr>';
                    return;
                }
                
                const courseData = courseDoc.data();
                const students = courseData.students || [];
                
                // Firebase se existing attendance fetch karein
                let existingAttendance = [];
                if (date) {
                    const attendanceQuery = query(
                        collection(db, "attendance"), 
                        where("courseId", "==", courseId),
                        where("date", "==", date)
                    );
                    
                    const attendanceSnapshot = await getDocs(attendanceQuery);
                    if (!attendanceSnapshot.empty) {
                        const attendanceData = attendanceSnapshot.docs[0].data();
                        existingAttendance = attendanceData.attendance || [];
                        console.log("Found existing attendance:", existingAttendance);
                    }
                }
                
                tbody.innerHTML = '';
                
                if (students.length === 0) {
                    tbody.innerHTML = '<tr><td colspan="4">No students in this course</td></tr>';
                    return;
                }
                
                // Har student ke liye row banayein aur existing attendance set karein
                students.forEach(student => {
                    const existingRecord = existingAttendance.find(
                        record => record.rollNumber === student.roll
                    );
                    
                    const isPresent = existingRecord ? 
                        (existingRecord.status === "Present") : 
                        true;
                    
                    const row = `<tr>
                        <td>${student.name || 'N/A'}</td>
                        <td>${student.roll || 'N/A'}</td>
                        <td>
                            <select class="att-status" style="background:#333; color:white;">
                                <option value="Present" ${isPresent ? 'selected' : ''}>Present</option>
                                <option value="Absent" ${!isPresent ? 'selected' : ''}>Absent</option>
                            </select>
                        </td>
                        <td><button onclick="this.parentElement.parentElement.remove()" style="color:red; background:none; border:none; cursor:pointer;">×</button></td>
                    </tr>`;
                    tbody.insertAdjacentHTML('beforeend', row);
                });
                
                console.log(`Loaded ${students.length} students with existing attendance`);
            } catch (error) {
                console.error("Error loading attendance:", error);
                tbody.innerHTML = '<tr><td colspan="4">Error loading attendance data</td></tr>';
            }
        };

        window.submitAttendance = async function() {
            const courseId = document.getElementById('attnSubject')?.value;
            const date = document.getElementById('attnDate')?.value;
            
            if (!courseId || !date) {
                alert('Please select course and date!');
                return;
            }
            
            try {
                const tableBody = document.getElementById('attendanceTableBody');
                const rows = tableBody.querySelectorAll('tr');
                
                const attendanceData = [];
                const studentsData = []; // Course ke students ke liye array
                
                rows.forEach(row => {
                    const nameInput = row.querySelector('.att-name');
                    const rollInput = row.querySelector('.att-roll');
                    const statusSelect = row.querySelector('.att-status');
                    
                    let studentName, rollNumber;
                    
                    // Agar input fields hain toh unse value lein, nahi toh text content se
                    if (nameInput && rollInput) {
                        studentName = nameInput.value.trim();
                        rollNumber = rollInput.value.trim();
                    } else {
                        const cells = row.querySelectorAll('td');
                        studentName = cells[0]?.textContent.trim();
                        rollNumber = cells[1]?.textContent.trim();
                    }
                    
                    if (studentName && rollNumber && statusSelect) {
                        attendanceData.push({
                            studentName: studentName,
                            rollNumber: rollNumber,
                            status: statusSelect.value,
                            date: date
                        });
                        
                        // Course students array ke liye bhi add karein
                        studentsData.push({
                            name: studentName,
                            roll: rollNumber
                        });
                    }
                });
                
                // Check if attendance already exists for this date
                const attendanceQuery = query(
                    collection(db, "attendance"), 
                    where("courseId", "==", courseId),
                    where("date", "==", date)
                );
                
                const attendanceSnapshot = await getDocs(attendanceQuery);
                
                if (!attendanceSnapshot.empty) {
                    // Update existing document
                    const docId = attendanceSnapshot.docs[0].id;
                    await setDoc(doc(db, "attendance", docId), {
                        courseId: courseId,
                        date: date,
                        attendance: attendanceData,
                        updatedAt: new Date().toISOString()
                    });
                    alert(`✅ Attendance updated for ${date}!`);
                } else {
                    // Create new document
                    const attendanceRef = doc(collection(db, "attendance"));
                    await setDoc(attendanceRef, {
                        courseId: courseId,
                        date: date,
                        attendance: attendanceData,
                        createdAt: new Date().toISOString()
                    });
                    alert(`✅ Attendance submitted for ${date}!`);
                }
                
                // Course mein students add/update karein
                if (studentsData.length > 0) {
                    const courseRef = doc(db, "courses", courseId);
                    await updateDoc(courseRef, {
                        students: studentsData,
                        studentsCount: studentsData.length
                    });
                }
                
                if (typeof showTeacherSection === 'function') {
                    showTeacherSection('dashboardView');
                }
            } catch (error) {
                console.error("Attendance submit error:", error);
                alert("Error submitting attendance!");
            }
        };

                    grid.innerHTML = '<div style="text-align:center; padding:40px; color:#ccc;">No courses available.</div>';
                }
            } catch (error) {
                console.error("Error loading student courses:", error);
                grid.innerHTML = '<div style="text-align:center; padding:40px; color:#e74c3c;">Error loading courses.</div>';
            }
        };

        window.viewCourse = async function(courseId) {
            try {
                const courseDoc = await getDoc(doc(db, "courses", courseId));
                if (courseDoc.exists()) {
                    const courseData = courseDoc.data();
                    
                    // Update student view
                    document.getElementById('detCourseName').textContent = courseData.courseName || "Course Name";
                    document.getElementById('detTeacherName').textContent = `Teacher: ${courseData.teacherName || "N/A"}`;
                    document.getElementById('presentPercent').textContent = courseData.presentPercent || "0%";
                    document.getElementById('absentPercent').textContent = courseData.absentPercent || "0%";
                    document.getElementById('syllabusContent').textContent = courseData.syllabus || "No syllabus available.";
                    document.getElementById('announcementContent').textContent = courseData.announcement || "No announcements.";
                    document.getElementById('lastUpdated').textContent = new Date(courseData.lastUpdated || new Date()).toLocaleDateString();
                    
                    // Update study materials
                    const downloadList = document.getElementById('studentDownloadList');
                    if (courseData.files && courseData.files.length > 0) {
                        downloadList.innerHTML = '';
                        courseData.files.forEach(file => {
                            const downloadItem = document.createElement('div');
                            downloadItem.className = 'download-item';
                            downloadItem.innerHTML = `
                                <span><i class="fas fa-link"></i> ${file.name}</span>
                                <a href="${file.url}" target="_blank" style="background:#cf9757; color:black; padding:5px 10px; border-radius:4px; text-decoration:none; font-size:0.8rem; font-weight:bold;">Open</a>
                            `;
                            downloadList.appendChild(downloadItem);
                        });
                    } else {
                        downloadList.innerHTML = '<p style="color: #ccc; text-align: center;">No study materials available</p>';
                    }
                    
                    // Show student course page
                    window.showStudentSection('studentCoursePageView');
                } else {
                    alert("Course not found!");
                }
            } catch (error) {
                console.error("Error viewing course:", error);
                alert("Error loading course details!");
            }
        };
